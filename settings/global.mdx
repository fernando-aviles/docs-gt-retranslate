## Overview

The `mint.json` file is the central configuration file for every Mintlify site. It controls everything from basic branding to complex navigation structures. All settings are configured using a JSON object with specific properties for different aspects of your documentation.

## Core Visual Settings

### Brand Identity

<ResponseField name="name" type="string" required>
  The name of your company or project that appears as the global title.
</ResponseField>

<ResponseField name="logo" type="string or Logo">
  Your brand logo configuration. SVG format is recommended for optimal quality and file size.

  <Expandable title="Logo Configuration">
    <ResponseField name="light" type="string">
      Logo path for light mode. Example: `/logo/light.svg`
    </ResponseField>

    <ResponseField name="dark" type="string">
      Logo path for dark mode. Example: `/logo/dark.svg`
    </ResponseField>

    <ResponseField name="href" type="string" default="/">
      Destination URL when clicking the logo
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="favicon" type="string">
  Path to your favicon image. Example: `/favicon.svg`
</ResponseField>

### Color Scheme

<ResponseField name="colors" type="Colors">
  Define your documentation's color palette.

  <Expandable title="Color Configuration">
    <ResponseField name="primary" type="string" required>
      Main brand color used for highlighting and primary UI elements in light mode
    </ResponseField>

    <ResponseField name="light" type="string">
      Primary color variant for dark mode UI elements
    </ResponseField>

    <ResponseField name="dark" type="string">
      Color used for important buttons and actions
    </ResponseField>

    <ResponseField name="background" type="object">
      Background colors for light and dark modes

      <Expandable title="Background Colors">
        <ResponseField name="light" type="string" required>
          Background color in light mode
        </ResponseField>

        <ResponseField name="dark" type="string" required>
          Background color in dark mode
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>

### Theme and Layout

<ResponseField name="theme" type={'"venus" | "quill" | "prism"'}>
  Choose from our preset theme configurations:
  - [Venus](https://starter-venus.mintlify.app)
  - [Quill](https://starter-quill.mintlify.app)
  - [Prism](https://starter-prism.mintlify.app)
</ResponseField>

<ResponseField name="layout" type={'"topnav" | "sidenav" | "solidSidenav"'} default="topnav">
  Select your documentation's primary navigation style
</ResponseField>

### Typography

<ResponseField name="font" type={'FontDetailsType | { headings?: FontDetailsType, body?: FontDetailsType }'}>
  Customize typography for your documentation.

Example:

```json
"font": {
  "headings": {
    "family": "Roboto"
  },
  "body": {
    "family": "Oswald"
  }
}
```

  <Expandable title="Font Configuration">
    <ResponseField name="family" type="string" required>
      Font family name - supports custom fonts and [Google Fonts](https://fonts.google.com/)
    </ResponseField>

    <ResponseField name="weight" type="number">
      Font weight, including variable font weights
    </ResponseField>

    <ResponseField name="url" type="string">
      Custom font file URL
    </ResponseField>

    <ResponseField name="format" type="'woff' | 'woff2'">
      Font format for custom fonts
    </ResponseField>
  </Expandable>
</ResponseField>

## Navigation Structure

### Main Navigation

<ResponseField name="navigation" type="Navigation[]" required>
  Define your documentation's page structure and hierarchy.

  <Expandable title="Navigation Structure">
    <ResponseField name="group" type="string">
      Section group name
    </ResponseField>

    <ResponseField name="pages" type="string[]">
      Array of page paths within the group
    </ResponseField>

    <ResponseField name="icon" type="string">
      [Fontawesome](https://fontawesome.com/icons) icon for sub-folders
    </ResponseField>

    <ResponseField name="iconType" type="string">
      Icon style: brands, duotone, light, sharp-solid, solid, thin
    </ResponseField>
  </Expandable>
</ResponseField>

### Top Navigation

<ResponseField name="topbarLinks" type="TopbarLink[]">
  Configure links in the top navigation bar.

  <Expandable title="Topbar Link">
    <ResponseField name="name" type="string">
      Link display text
    </ResponseField>

    <ResponseField name="url" type="string">
      Link destination URL
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="topbarCtaButton" type="Call to Action">
  Configure the call-to-action button in top navigation.

  <Expandable title="CTA Configuration">
    <ResponseField name="type" type={"link or github"} default="link">
      Button type - standard link or GitHub repository link
    </ResponseField>

    <ResponseField name="url" type="string">
      Button destination URL
    </ResponseField>

    <ResponseField name="name" type="string">
      Button text for link type
    </ResponseField>

    <ResponseField name="style" type={'"pill" | "roundedRectangle"'}>
      Button visual style
    </ResponseField>

    <ResponseField name="arrow" type="boolean">
      Show/hide button arrow
    </ResponseField>
  </Expandable>
</ResponseField>

### Versioning and Localization

<ResponseField name="versions" type="string[]">
  Configure version dropdowns and localization support.

<Info>
  For localization, UI elements like "Was this page helpful?" are automatically translated.
  You must translate page content manually.
</Info>

Example configurations:

  <CodeGroup>
    ```json Version Example
    "versions": ["v1.0", "v1.1"]
    ```

    ```json Localization Example
    "versions": [
      {
        "name": "English",
        "locale": "en",
      },
      {
        "name": "Espa√±ol",
        "locale": "es"
      }
    ]
    ```
  </CodeGroup>

  <Expandable title="Version Properties">
    <ResponseField name="name" type="string" required>
      Version identifier
    </ResponseField>

    <ResponseField name="locale" type="string">
      Language code (en, cn, es, fr, jp, pt, pt-BR, de)
    </ResponseField>

    <ResponseField name="default" type="true">
      Set as default version
    </ResponseField>
  </Expandable>
</ResponseField>

## Enhanced Navigation Features

### Anchors

<ResponseField name="anchors" type="Anchor[]">
  Configure navigational anchors with custom styling.

  <Expandable title="Anchor Properties">
    <ResponseField name="name" type="string" required>
      Anchor label name
    </ResponseField>

    <ResponseField name="icon" type="string">
      [Font Awesome](https://fontawesome.com/search?q=heart) icon identifier
    </ResponseField>

    <ResponseField name="url" type="string">
      Section URL path
    </ResponseField>

    <ResponseField name="color" type="string">
      Icon background color (hex or gradient object)
    </ResponseField>

    <ResponseField name="version" type="string">
      Version-specific visibility
    </ResponseField>

    <ResponseField name="isDefaultHidden" type="boolean" default={false}>
      Hide until directly linked
    </ResponseField>
  </Expandable>
</ResponseField>

### Tabs

<ResponseField name="tabs" type="Tabs[]">
  Create tabbed navigation sections.

Example:
```json
"tabs": [
  {
    "name": "Writing Content",
    "url": "content"
  },
  {
    "name": "API References",
    "url": "api-playground"
  }
]
```

  <Expandable title="Tab Properties">
    <ResponseField name="name" type="string">
      Tab label text
    </ResponseField>

    <ResponseField name="url" type="string">
      Tab content path
    </ResponseField>

    <ResponseField name="isDefaultHidden" type="boolean" default={false}>
      Hide until directly accessed
    </ResponseField>
  </Expandable>
</ResponseField>

## Additional Features

### Footer Configuration

<ResponseField name="footer" type="{ socials?: FooterSocials, links?: FooterLinksColumn[] }">
  Configure footer content and social links.

Example:
```json
"footer": {
  "socials": {
    "x": "https://x.com/mintlify",
    "website": "https://mintlify.com"
  },
  "links": [
    {
      "title": "Column 1",
      "links": [
        {
          "label": "Link 1",
          "url": "https://mintlify.com"
        }
      ]
    }
  ]
}
```

  <Expandable title="Footer Properties">
    <ResponseField name="[key]" type="string">
      Social platform identifiers
    </ResponseField>

    <ResponseField name="title" type="string">
      Column header text
    </ResponseField>

    <ResponseField name="links" type="{ label: string, url: string }[]">
      Footer link items
    </ResponseField>
  </Expandable>
</ResponseField>

### User Feedback

<ResponseField name="feedback" type="Feedback">
  Enable user feedback features.

  <Expandable title="Feedback Options">
    <ResponseField name="thumbsRating" type="boolean" default={false}>
      Enable helpful/not helpful rating
    </ResponseField>

    <ResponseField name="suggestEdit" type="boolean" default={false}>
      Enable edit suggestions via PR
    </ResponseField>

    <ResponseField name="raiseIssue" type="boolean" default={false}>
      Enable issue reporting
    </ResponseField>
  </Expandable>
</ResponseField>

## API Integration
