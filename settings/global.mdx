Every Mintlify site requires a `mint.json` file for core configuration settings.

## Properties

### Core Styling

<ResponseField name="name" type="string" required>
  Name of your company or project (used for global title)
</ResponseField>

<ResponseField name="logo" type="string or Logo">
  Path to logo image or object with "light" and "dark" mode logo paths. SVG recommended.

  <Expandable title="Logo">
    <ResponseField name="light" type="string">Path to light mode logo</ResponseField>
    <ResponseField name="dark" type="string">Path to dark mode logo</ResponseField>
    <ResponseField name="href" type="string" default="/">Logo click destination</ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="favicon" type="string">
  Path to favicon image
</ResponseField>

<ResponseField name="colors" type="Colors">
  Theme color configuration

  <Expandable title="Colors">
    <ResponseField name="primary" type="string" required>Primary theme color</ResponseField>
    <ResponseField name="light" type="string">Primary color for dark mode</ResponseField>
    <ResponseField name="dark" type="string">Color for important buttons</ResponseField>
    <ResponseField name="background" type="object">
      <Expandable title="Object">
        <ResponseField name="light" type="string" required>Light mode background</ResponseField>
        <ResponseField name="dark" type="string" required>Dark mode background</ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>

### Layout & Theme

<ResponseField name="theme" type={'"venus" | "quill" | "prism"'}>
  Preset theme configuration ([Venus](https://starter-venus.mintlify.app), [Quill](https://starter-quill.mintlify.app), [Prism](https://starter-prism.mintlify.app))
</ResponseField>

<ResponseField name="layout" type={'"topnav" | "sidenav" | "solidSidenav"'} default="topnav">
  Global layout style
</ResponseField>

### Navigation

<ResponseField name="navigation" type="Navigation[]" required>
  Page structure and grouping

  <Expandable title="Navigation">
    <ResponseField name="group" type="string">Group name</ResponseField>
    <ResponseField name="pages" type="string[]">Paths to markdown files</ResponseField>
    <ResponseField name="icon" type="string">Fontawesome icon (for sub-folders)</ResponseField>
    <ResponseField name="iconType" type="string">Icon type (brands, duotone, light, sharp-solid, solid, thin)</ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="topbarLinks" type="TopbarLink[]">
  Header navigation links
  <Expandable title="TopbarLink">
    <ResponseField name="name" type="string">Button name</ResponseField>
    <ResponseField name="url" type="string">Target URL</ResponseField>
  </Expandable>
</ResponseField>

### API Configuration

<ResponseField name="api" type="API">
  <Expandable title="API">
    <ResponseField name="baseUrl" type="string">Base URL for API endpoints</ResponseField>
    <ResponseField name="auth" type="Auth">Authentication settings</ResponseField>
    <ResponseField name="playground" type="Playground">API playground configuration</ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="openapi" type="string | string[]">
  OpenAPI specification URL(s) or path(s)
</ResponseField>

### Additional Features

<ResponseField name="feedback" type="Feedback">
  Enable user feedback features (thumbs rating, edit suggestions, issue reporting)
</ResponseField>

<ResponseField name="analytics" type="Analytics">
  Third-party analytics integrations
</ResponseField>

<ResponseField name="integrations" type="Integrations">
  Third-party integrations (Intercom, Frontchat, etc.)
</ResponseField>

<ResponseField name="redirects" type="Redirect[]">
  URL redirect configurations
</ResponseField>

<ResponseField name="seo" type="SEO">
  Search engine optimization settings
</ResponseField>

## Example Configuration

<Accordion title="View Example Configuration">
```json
{
  "name": "Mintlify",
  "logo": {
    "light": "/logo/light.svg",
    "dark": "/logo/dark.svg"
  },
  "favicon": "/favicon.svg",
  "colors": {
    "primary": "#16A34A",
    "light": "#4ADE80",
    "dark": "#166534"
  },
  "navigation": [
    {
      "group": "Getting Started",
      "pages": ["introduction", "quickstart"]
    }
  ],
  "topbarLinks": [
    {
      "name": "Contact Us",
      "url": "mailto:support@mintlify.com"
    }
  ],
  "api": {
    "baseUrl": "https://api.example.com",
    "playground": {
      "mode": "show"
    }
  }
}
```
</Accordion>

For advanced customization options, see [Custom Scripts](https://mintlify.com/docs/advanced/custom/).
