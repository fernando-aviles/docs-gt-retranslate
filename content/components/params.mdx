`ParamField` components define API/SDK parameters and automatically add an [API Playground](/api-playground/overview).

<RequestExample>

```jsx Example
<ParamField path="param" type="string" required>
  Parameter field example
</ParamField>

<ParamField query="filter" type="string" default="none" required>
  Filtering command for user sorting
</ParamField>

<ParamField body="user_age" type="integer" default="0" required>
  User age (must be >= 0)
</ParamField>
```

</RequestExample>

## Props

<ParamField body="query, path, body, or header" type="string">
  Parameter type and name
</ParamField>

<ParamField body="type" type="string">
  Expected value type
</ParamField>

<ParamField body="required" type="boolean">
  Whether parameter is required
</ParamField>

<ParamField body="deprecated" type="boolean">
  Whether parameter is deprecated
</ParamField>

<ParamField body="default" type="string">
  Server's default value if none provided
</ParamField>

<ParamField body="initialValue" type="any">
  Playground initialization value
</ParamField>

<ParamField body="placeholder" type="string">
  Input placeholder text
</ParamField>

<ParamField body="children" type="string">
  Parameter description (markdown enabled)
</ParamField>
