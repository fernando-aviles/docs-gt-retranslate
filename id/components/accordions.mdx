### Props

<ResponseField name="title" type="string" required>
  Judul dalam pratinjau Accordion.
</ResponseField>

<ResponseField name="description" type="string">
  Detail di bawah judul dalam pratinjau Accordion.
</ResponseField>

<ResponseField name="defaultOpen" type="boolean" default="false">
  Apakah Accordion terbuka secara default.
</ResponseField>

<ResponseField name="icon" type="string or svg">
  Sebuah[Font Awesome icon](https://fontawesome.com/icons),[Lucide
  icon](https://lucide.dev/icons), atau kode SVG
</ResponseField>

<ResponseField name="iconType" type="string">
  Salah satu dari "regular", "solid", "light", "thin", "sharp-solid", "duotone", atau
  "brands"
</ResponseField>
