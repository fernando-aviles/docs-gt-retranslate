## Props

<ResponseField name="title" type="string" required>
  Título en la vista previa del Acordeón.
</ResponseField>

<ResponseField name="description" type="string">
  Detalle debajo del título en la vista previa del Acordeón.
</ResponseField>

<ResponseField name="defaultOpen" type="boolean" default="false">
  Si el Acordeón está abierto por defecto.
</ResponseField>

<ResponseField name="icon" type="string or svg">
  Un [ícono de Font Awesome](https://fontawesome.com/icons), [ícono de Lucide](https://lucide.dev/icons), o código SVG
</ResponseField>

<ResponseField name="iconType" type="string">
  Uno de "regular", "solid", "light", "thin", "sharp-solid", "duotone", o "brands"
</ResponseField>
